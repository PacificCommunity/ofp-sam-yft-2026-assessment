\begin{frame}{External tagging analysis}\small
  Related but different from SEAPODYM\\[4ex]
  \textblue{\bf DTU Movement Model}
  \begin{itemize}
    \item[] Uses all tags, also those that are not recaptured\\[-0.6ex]
    \item[] Designed to produce abundance indices\\[-0.6ex]
    \item[] Software maintained by Tobias Mildenberger (DTU) using RTMB\\[4ex]
  \end{itemize}
  \textgreen{\bf SEAPODYM}
  \begin{itemize}
    \item[] Only uses recaptured tags\\[-0.6ex]
    \item[] Designed to estimate fish movement\\[-0.6ex]
    \item[] Software maintained by Inna Senina (SPC) using AUTODIF \cpp\
    libraries
  \end{itemize}
\end{frame}
